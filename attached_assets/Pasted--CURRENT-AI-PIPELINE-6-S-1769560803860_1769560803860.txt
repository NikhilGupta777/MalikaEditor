1. VIDEO UPLOAD → Extract 6-12 frames (snapshots only - NO full video watching)
                  Extract audio file
                  Detect silent segments
2. TRANSCRIPTION → AssemblyAI (primary) with speakers, chapters, sentiment
                   OR OpenAI/Gemini fallback
                   Word-level timestamps for captions
3. FRAME ANALYSIS → Send 6-12 JPEG images to Gemini
                    AI describes each frame independently
                    Guesses scene transitions, emotion flow
                    ⚠️ LIMITATION: Cannot see actual motion/transitions
4. SEMANTIC ANALYSIS → Analyze transcript text
                       Identify topics, tone, hooks, structure
                       Generate B-roll queries
5. EDIT PLANNING → 2-pass AI planning
                   Pass 1: Structure + Quality + B-roll
                   Pass 2: Quality review
                   Generates: cuts, keeps, B-roll windows, caption timing
6. MEDIA SELECTION → Fetch stock from Pexels + Freepik
                     Generate AI images
                     ✅ NEW: Analyze thumbnails visually
                     AI selects best media for each window
7. USER REVIEW → User approves/rejects decisions
                 Pre-render AI review for confidence scoring
8. RENDER → FFmpeg applies all edits
            ⚠️ NO AI watches the final output

// New capability: Upload and watch entire video
async function watchFullVideo(videoPath: string): Promise<DeepVideoAnalysis> {
  const gemini = getGeminiClient();
  
  // Upload video file to Gemini
  const videoFile = await gemini.files.upload({
    file: videoPath,
    mimeType: "video/mp4"
  });
  
  // Wait for processing
  await waitForFileProcessing(videoFile.uri);
  
  // AI watches entire video and provides comprehensive analysis
  const response = await gemini.models.generateContent({
    model: "gemini-2.5-flash",
    contents: [
      { fileData: { fileUri: videoFile.uri } },
      { text: COMPREHENSIVE_VIDEO_ANALYSIS_PROMPT }
    ]
  });
  
  return parseVideoAnalysis(response);
}

STAGE 2: Self-Review System (HIGH PRIORITY)

AI watches its own rendered output and evaluates quality

interface SelfReviewResult {
  overallScore: number; // 0-100
  categoryScores: {
    pacing: number;
    transitions: number;
    brollRelevance: number;
    captionTiming: number;
    audioQuality: number;
    goalAchievement: number;
  };
  issues: {
    timestamp: number;
    type: "bad_cut" | "wrong_broll" | "caption_issue" | "pacing_issue";
    description: string;
    suggestedFix: string;
  }[];
  passesThreshold: boolean;
}
async function reviewOwnOutput(
  outputVideoPath: string,
  originalAnalysis: VideoAnalysis,
  userPrompt: string,
  editPlan: EditPlan
): Promise<SelfReviewResult> {
  // Upload rendered video for AI to watch
  const videoFile = await gemini.files.upload({ file: outputVideoPath });
  
  // AI reviews its own work
  const response = await gemini.models.generateContent({
    model: "gemini-2.5-flash",
    contents: [
      { fileData: { fileUri: videoFile.uri } },
      { text: buildSelfReviewPrompt(originalAnalysis, userPrompt, editPlan) }
    ]
  });
  
  return parseSelfReview(response);
}

STAGE 3: Iterative Correction Loop (HIGH PRIORITY)

Re-edit based on self-review feedback

async function autonomousEditLoop(
  projectId: string,
  maxIterations: number = 3
): Promise<string> {
  let iteration = 0;
  let finalVideoPath: string;
  
  while (iteration < maxIterations) {
    // Render current edit plan
    finalVideoPath = await renderVideo(projectId);
    
    // AI reviews its own output
    const review = await reviewOwnOutput(finalVideoPath, ...);
    
    if (review.passesThreshold) {
      logger.info(`Quality threshold passed on iteration ${iteration + 1}`);
      return finalVideoPath;
    }
    
    // Generate corrective edit plan based on issues
    const corrections = await generateCorrections(review.issues, editPlan);
    
    // Apply corrections and re-render
    await applyCorrections(projectId, corrections);
    
    iteration++;
  }
  
  return finalVideoPath;
}

STAGE 4: Learning System (MEDIUM PRIORITY)

Learn from user feedback and successful patterns

interface EditingPattern {
  context: {
    genre: string;
    tone: string;
    duration: number;
  };
  successfulDecisions: {
    cutPatterns: CutPattern[];
    brollStrategies: BrollStrategy[];
    pacingChoices: PacingChoice[];
  };
  userApprovalRate: number;
}
// Store patterns from successful edits
async function learnFromSuccess(
  projectId: string,
  userApproved: boolean,
  finalReviewScore: number
): Promise<void> {
  if (userApproved && finalReviewScore > 85) {
    const patterns = extractPatterns(projectId);
    await storage.storeSuccessfulPattern(patterns);
  }
}
// Use learned patterns in future edits
async function applyLearnedPatterns(
  videoContext: VideoContext
): Promise<EditingHints> {
  const similarPatterns = await storage.findSimilarPatterns(videoContext);
  return generateHintsFromPatterns(similarPatterns);
}
